## Controlled Dropdown

<Playground>
  <Component initialState={{ value: 'ALL', isOpen: false }}>
    {({ setState, state }) => (
      <Row>
        <Dropdown
          isOpen={state.isOpen}
          onItemSelect={(option, e) => setState({ value: option.value })}
          toggle={(isOpen) => setState({ isOpen })}
          options={[
            { name: 'ALL', value: 'ALL' },
            { name: 'Cash Deposits', value: 'FIAT_DEPOSIT' },
            { name: 'Cash Withdrawals', value: 'FIAT_WITHDRAWAL', active: true },
            { name: 'separator-1', value: 'SEPARATOR' },
            { name: 'Token Deposits', value: 'TOKEN_DEPOSIT', disabled: true },
            { name: 'Token Withdrawals', value: 'TOKEN_WITHDRAWAL' },
          ]}
          styleMode="dark"
          value={state.value}
        />
         <Button
            btnType="primary"
            onClick={() => setState({ isOpen: !state.isOpen })}
            styleMode="dark"
            size="lg"
          >
            {state.isOpen ? 'Close' : 'Open'} me
          </Button>
      </Row>
    )}
  </Component>
</Playground>
