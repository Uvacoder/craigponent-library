## Controlled Alert
* Wrap a `Fade` component around for a smooth transition
* You can also Wrap a `Collapse` component around that `Fade` for a smoother transition

<Playground>
  <Component
    initialState={{
      isOpen1: true,
      isOpen2: true,
      isOpen3: true,
      fade1: true,
      fade2: true,
    }}
  >
    {({ setState, state }) => (
      <>
        <Alert
          color="primary"
          isOpen={state.isOpen1}
          onClose={() => setState({ isOpen1: false})}
        >
          This primary alert can be closed/controlled
        </Alert>
        <Fade in={state.fade1} duration={300} onExited={() => setState({ isOpen2: false })}>
          <Alert
            color="success"
            isOpen={state.isOpen2}
            onClose={() => setState({ fade1: false })}
          >
            This success alert has a fade animation
          </Alert>
        </Fade>
        <Collapse isOpen={state.isOpen3}>
          <Fade in={state.fade2} duration={300} onExited={() => setState({ isOpen3: false })}>
            <Alert
              color="danger"
              onClose={() => setState({ fade2: false })}
            >
              This danger alert has a fade and collapse animation
            </Alert>
          </Fade>
        </Collapse>
        <Button
          onClick={() => setState({
            isOpen1: true,
            isOpen2: true,
            isOpen3: true,
            fade1: true,
            fade2: true
          })}
        >
          Refresh
        </Button>
      </>
    )}
  </Component>
</Playground>
