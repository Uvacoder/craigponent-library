# ProgressTracker

ProgressTracker shows an easy status update on which page the user is on. 1 - 2 - 3

```javascript
import ProgressTracker from 'craig-component-library/dist/ProgressTracker'
```

## Properties

<PropsTable of={ProgressTracker} />


## Basic Usage

<Playground>
  <Component
    initialState={{
      activeIndex: 1
    }}
  >
    {({ setState, state }) => {
      const progressTrackerWords = [
        'Create Account',
        'Issuer Info',
        'Offering Info',
        'Submit',
      ];
      return (
        <>
          <ProgressTracker
            activeIndex={state.activeIndex}
            selectIndex={(i) => () => setState({ activeIndex: i })}
            items={progressTrackerWords}
          />

          <div style={{ display: 'flex', justifyContent: 'space-around', marginTop: '100px' }}>
            <Button
              disabled={state.activeIndex === 1}
              onClick={() => setState({ activeIndex: state.activeIndex - 1 })}
            >
              Previous Page
            </Button>
            <Button
              disabled={progressTrackerWords.length === state.activeIndex}
              onClick={() => setState({ activeIndex: state.activeIndex + 1 })}
            >
              Next Page
            </Button>
          </div>
        </>
      )
    }}
  </Component>
</Playground>
