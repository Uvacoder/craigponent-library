## Controlled Alert

<Playground>
  <Component
    initialState={{
      isOpen: true
    }}
  >
    {({ setState, state }) => (
      <>
        <Alert
          color="danger"
          isOpen={state.isOpen}
          onClose={() => setState({ isOpen: false})}
        >
          This danger alert can be closed/controlled
        </Alert>
        <Button onClick={() => setState({ isOpen: true })}>Refresh</Button>
      </>
    )}
  </Component>
</Playground>
